28 inline typename MatrixT::ValueType Determinant( const MatrixT& i_matrix )
29 {
30  static_assert( MatrixT::RowCount() == MatrixT::ColumnCount() );
31 
32  // Left-hand-side working matrix. This will assume the identity matrix after the full Gauss-Jordan elimination.
33  MatrixT matrix = i_matrix;
34 
35  // Use cache to record on i_matrix.
36  MatrixEntryArray< MatrixT::RowCount(), typename MatrixT::ValueType > eliminationFactors;
37 
38  // Count the number of row exchanges.
39  int numRowExchanges = 0;
40 
41  for ( int pivotIndex = 0; pivotIndex < MatrixT::RowCount() - 1; ++pivotIndex )
42  {
43  // Check that the current pivot is non-zero.
44  if ( matrix( pivotIndex, pivotIndex ) == 0 )
45  {
46  // Try to find a row below with a non-zero pivot to exchange.
47  int exchangedRow = _FindAndPerformRowExchange( pivotIndex, pivotIndex, matrix );
48  if ( exchangedRow != -1 )
49  {
50  numRowExchanges++;
51  }
52  else
53  {
54  // Matrix is singular, then the determinant is 0.
55  return 0;
56  }
57  }
58 
59  // Perform elimination.
60  _RecordElimination( pivotIndex,
61  pivotIndex,
62  IntRange( pivotIndex + 1, MatrixT::RowCount() ) /* rowRange */,
63  IntRange( pivotIndex, MatrixT::ColumnCount() ) /* columnRange */,
64  eliminationFactors,
65  matrix );
66  }
67 
68  // Compute product of pivots.
69  typename MatrixT::ValueType determinant = matrix( 0, 0 );
70  for ( int pivotIndex = 1; pivotIndex < MatrixT::RowCount(); ++pivotIndex )
71  {
72  determinant *= matrix( pivotIndex, pivotIndex );
73  }
74 
75  // Odd # of row exchanges imparts a -1 factor.
76  if ( numRowExchanges % 2 == 0 )
77  {
78  return determinant;
79  }
80  else
81  {

83  }
